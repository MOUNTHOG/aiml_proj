% Rules
% parent is either father or mother
parent(X, Y) :- father(X, Y).
parent(X, Y) :- mother(X, Y).
% siblings share at least one parent
sibling(X, Y) :-
parent(P, X),
parent(P, Y),
X \= Y.
% brother if male sibling
brother(X, Y) :-
sibling(X, Y),
male(X).
% sister if female sibling
sister(X, Y) :-
sibling(X, Y),
female(X).
